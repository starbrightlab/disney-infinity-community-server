# Disney Infinity Community Server - Hosts File Override
# Add these entries to your system's hosts file to redirect Disney UGC traffic
# to the community server.
#
# Windows: C:\Windows\System32\drivers\etc\hosts
# Linux/macOS: /etc/hosts
#
# Replace YOUR_SERVER_IP with your actual server IP address or domain

# Disney UGC and Toybox Services
YOUR_SERVER_IP disney.go.com
YOUR_SERVER_IP toys.disney.go.com
YOUR_SERVER_IP ugc.disney.go.com
YOUR_SERVER_IP api.toybox.com
YOUR_SERVER_IP toybox.disney.go.com

# Authentication Services
YOUR_SERVER_IP login.disney.go.com

# Matchmaking Services (if implemented)
YOUR_SERVER_IP matchmaking.disney.go.com

# CDN and Asset Services
YOUR_SERVER_IP cdn.disney.go.com
YOUR_SERVER_IP assets.disney.go.com
